    username VARCHAR(50),

);

CREATE TABLE courses(
    course_id INT PRIMARY KEY,
    course_name VARCHAR(100)
);

CREATE TABLE students(
    student_id INT AUTO_INCREMENT PRIMARY KEY,
    name VARCHAR(100),
    username VARCHAR(50),
    password VARCHAR(50)
);

CREATE TABLE enrollments(
    student_id INT,
    course_id INT
);
